function solution(number){
  if (number < 0) return 0

  let array = []

  for (let i = 1; i < number; i++) {
    if (i % 3 == 0 || i % 5 == 0) {
      array.push(i)
    }
  }

  let sum = 0
  for (let j = 0; j < array.length; j++) {
    sum = sum + array[j]
  }
  return sum
}
